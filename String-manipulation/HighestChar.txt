//HighestChar

//DECLARATION
DECLARE s:STRING
DECALRE alphabets: STRING
DECLARE HighestCount: INTEGER
DECLARE count: INTEGER
DECLARE mychar: CHAR
DECLARE HighestChar: CHAR
DECLARE x: INTEGER
DECLARE y: INTEGER


//INITIALIZATION
s=""
alphabets="abcdefghijklmnopqrstuvwxyz"
HighestCount=-1000
count=0
mychar=''
HighestChar=''
x=0
y=0

PRINT"Input a string: "
INPUT s
s=LOWER(s)
FOR x=1 TO 26
    count=0
    mychar=MID(alphabets,x,1)
    y=InStr(s,mychar)
    WHILE y>0
        count=count+1
        y=y+1
        y=Locate(y,s,mychar)
    END WHILE
    IF count>HighestCount
        HighestCount=count
        HighestChar=mychar
    END IF
NEXT X
PRINT"The most occuring alphabet is: " & HighestChar
PRINT"The most occuring alphabet is: " & HighestCount
