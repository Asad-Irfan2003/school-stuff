//DECLARATION
DECLARE x: STRING
DELCARE vowel: STRING
DELCARE vowelcount: INTEGER
DECLARE z: INTEGER
DELCARE y: INTEGER
DECLARE mychar: CHAR

//INITIALIZATION
x <- ''
vowel <- "aeiou"
vowelcount <- 0
middleStr <- ''

//WORKING TO FIND NUMBER OF VOWELS
INPUT"input a string: ",x
FOR z=1 TO LEN(x)
  FOR y=1 to 5
    mychar=MID(x,z,1)
    IF mychar=MID(vowel,y,1)
        vowelcount=vowelcount+1
    END IF
  NEXT y
NEXT z
